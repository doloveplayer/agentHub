```typescript
// apps/api/src/agent/InboxManager.ts
import { readFileSync, writeFileSync, existsSync, mkdirSync } from 'fs';
import { resolve } from 'path';

export interface InboxMessage {
  type: 'intervention_request' | 'intervention_response';
  id: string;
  from: string;
  to: string;
  summary?: string;
  risk?: 'low' | 'high';
  accepted?: boolean;
  message?: string;
  timestamp: number;
}

export class InboxManager {
  /**
   * Append a message to an agent's inbox file.
   * Inbox lives in the shared sandbox host directory:
   *   .sandboxes/{sessionId}/_inbox_{agentMessageId}.jsonl
   */
  static send(hostWorkDir: string, toAgentMessageId: string, msg: InboxMessage): void {
    const inboxPath = resolve(hostWorkDir, `_inbox_${toAgentMessageId}.jsonl`);
    const line = JSON.stringify(msg) + '\n';
    writeFileSync(inboxPath, line, { flag: 'a' });
  }

  /**
   * Read all unread messages from an agent's inbox.
   * Returns messages and optionally marks them as read by truncating.
   */
  static read(hostWorkDir: string, agentMessageId: string, markRead = true): InboxMessage[] {
    const inboxPath = resolve(hostWorkDir, `_inbox_${agentMessageId}.jsonl`);
    if (!existsSync(inboxPath)) return [];

    const content = readFileSync(inboxPath, 'utf-8');
    const messages: InboxMessage[] = [];
    for (const line of content.split('\n')) {
      if (!line.trim()) continue;
      try { messages.push(JSON.parse(line)); } catch { /* skip malformed */ }
    }

    if (markRead) {
      // Truncate file after read
      writeFileSync(inboxPath, '');
    }

    return messages;
  }

  /** Check if inbox has pending messages (non-destructive) */
  static hasPending(hostWorkDir: string, agentMessageId: string): boolean {
    const inboxPath = resolve(hostWorkDir, `_inbox_${agentMessageId}.jsonl`);
    return existsSync(inboxPath) && readFileSync(inboxPath, 'utf-8').trim().length > 0;
  }

  /** Clean up inbox file when agent finishes */
  static cleanup(hostWorkDir: string, agentMessageId: string): void {
    const inboxPath = resolve(hostWorkDir, `_inbox_${agentMessageId}.jsonl`);
    try { if (existsSync(inboxPath)) writeFileSync(inboxPath, ''); } catch { /* ignore */ }
  }
}
```

### Step 1: 实现 MilestoneBroadcaster

```typescript
// apps/api/src/agent/MilestoneBroadcaster.ts
import type { ParsedEvent } from './EventParser.js';

export type MilestoneLevel = 'internal' | 'milestone' | 'blocked';

export interface MilestoneEvent {
  level: MilestoneLevel;
  agentMessageId: string;
  agentId: string;
  sessionId: string;
  summary: string;
  details?: Record<string, unknown>;
  timestamp: number;
}

/**
 * Classify a raw Claude Code event into a broadcast level.
 *
 * Rules:
 *   internal  — thinking, tool_use, tool_result → only Agent Card
 *   milestone — Write/Edit tool_result with file path → group chat
 *   blocked   — agent hasn't produced output for > 30s while waiting
 */
export class MilestoneBroadcaster {
  private lastActivity = new Map<string, number>();

  classify(
    event: ParsedEvent,
    agentMessageId: string,
    agentId: string,
    sessionId: string,
  ): MilestoneLevel {
    this.lastActivity.set(agentMessageId, Date.now());

    switch (event.type) {
      case 'tool_result': {
        // If the tool was Write/Edit and produced a file, it's a milestone
        const content = typeof event.content === 'string' ? event.content : '';
        if (content.includes('/workspace/') || content.includes('file')) {
          return 'milestone';
        }
        return 'internal';
      }

      case 'done':
        return 'milestone';  // Agent finished → broadcast to group

      case 'error':
        return 'blocked';    // Agent errored → broadcast for help

      default:
        return 'internal';
    }
  }

  /** Check if agent is stalled (no activity in 30s) */
  checkStalled(agentMessageId: string): boolean {
    const last = this.lastActivity.get(agentMessageId);
    return last ? (Date.now() - last) > 30_000 : false;
  }

  buildMilestone(
    event: ParsedEvent,
    agentMessageId: string,
    agentId: string,
    sessionId: string,
    level: MilestoneLevel,
  ): MilestoneEvent {
    let summary = '';
    switch (event.type) {
      case 'tool_result':
        summary = `产出文件: ${(event as any).content?.slice(0, 200) || '完成'}`;
        break;
      case 'done':
        summary = `Agent 任务完成 (exit ${event.exitCode})`;
        break;
      case 'error':
        summary = `Agent 遇到错误: ${event.message?.slice(0, 100)}`;
        break;
      default:
        summary = event.type;
    }

    return {
      level,
      agentMessageId,
      agentId,
      sessionId,
      summary,
      details: event.type === 'tool_result' ? { content: (event as any).content } : undefined,
      timestamp: Date.now(),
    };
  }
}
```

### Step 2: 在 handler.ts onEvent 中接入

在 `onEvent` 回调的最后（每个 case 之后），调用 `classify()`。如果 `level === 'milestone' || level === 'blocked'`，执行 `broadcast(sessionId, { type: 'milestone', ... })` 发送到群聊。

### Implementation

在 `handleChatMessage` 中检测 mention 间的 `->` 符号：

```typescript
// 解析顺序编排: @A promptA -> @B promptB
interface SequentialMention {
  agentId: string;
  subPrompt: string;
  messageId: string;
  waitFor?: string;  // messageId of agent to wait for
}

function parseSequentialMentions(
  content: string,
  mentions: { agentId: string; subPrompt: string; messageId: string }[],
): SequentialMention[] {
  const parts = content.split('->');
  if (parts.length < 2) return mentions.map(m => ({ ...m }));

  const seq: SequentialMention[] = [];
  for (let i = 0; i < mentions.length; i++) {
    seq.push({
      ...mentions[i],
      waitFor: i > 0 ? mentions[i - 1].messageId : undefined,
    });
  }
  return seq;
}
```

当 `waitFor` 存在时，不在当前循环中启动该 Agent，而是注册一个监听器：当 `waitFor` 指向的 Agent 发送 `milestone`（done 事件）后，自动启动等待的 Agent。

```typescript
// 延迟启动队列
const pendingAgents = new Map<string, SequentialMention[]>();

// 当 Agent A done 时，检查是否有等待它的 Agent B
if (pendingAgents.has(mention.messageId)) {
  const waiting = pendingAgents.get(mention.messageId)!;
  for (const sm of waiting) {
    // 启动 Agent B，prompt 中注入 Agent A 的产出摘要
    startAgentForMention(sessionId, sm, sandbox, accumulatedContent);
  }
  pendingAgents.delete(mention.messageId);
}
```

### Step 1: 实现 WorkspaceManager

```typescript
// apps/api/src/agent/WorkspaceManager.ts
import { execSync } from 'child_process';

export interface WorkspaceSnapshot {
  type: 'stash' | 'branch';
  ref: string;  // stash hash or branch name
  workspacePath: string;
  createdAt: number;
}

export class WorkspaceManager {
  /** Create a git snapshot before agent execution */
  static snapshot(workspacePath: string): WorkspaceSnapshot | null {
    try {
      // Check if git repo
      execSync('git rev-parse --git-dir', { cwd: workspacePath, stdio: 'pipe' });

      // Create stash snapshot (lighter than branch)
      const ref = execSync('git stash create', {
        cwd: workspacePath,
        encoding: 'utf-8',
        stdio: 'pipe',
      }).trim();

      if (!ref) return null;  // no changes to snapshot

      return {
        type: 'stash',
        ref,
        workspacePath,
        createdAt: Date.now(),
      };
    } catch {
      // Not a git repo → fallback: create a tar backup
      const timestamp = Date.now();
      const tarPath = `${workspacePath}/.agenthub_backup_${timestamp}.tar.gz`;
      try {
        execSync(`tar -czf ${tarPath} -C ${workspacePath} . --exclude='.agenthub_backup_*'`, { stdio: 'pipe' });
        return { type: 'stash', ref: tarPath, workspacePath, createdAt: timestamp };
      } catch {
        return null;
      }
    }
  }

  /** Restore workspace to snapshot state */
  static rollback(snapshot: WorkspaceSnapshot): boolean {
    try {
      if (snapshot.type === 'stash' && !snapshot.ref.startsWith('/')) {
        // Git stash ref
        execSync(`git stash apply ${snapshot.ref}`, {
          cwd: snapshot.workspacePath,
          stdio: 'pipe',
        });
        return true;
      } else if (snapshot.type === 'stash') {
        // Tar fallback
        execSync(`tar -xzf ${snapshot.ref} -C ${snapshot.workspacePath}`, { stdio: 'pipe' });
        return true;
      }
      return false;
    } catch {
      return false;
    }
  }

  /** Clean up snapshot (delete stash or tar file) */
  static cleanup(snapshot: WorkspaceSnapshot): void {
    try {
      if (snapshot.ref.startsWith('/')) {
        execSync(`rm -f ${snapshot.ref}`, { stdio: 'pipe' });
      } else {
        execSync(`git stash drop ${snapshot.ref}`, {
          cwd: snapshot.workspacePath,
          stdio: 'pipe',
        });
      }
    } catch { /* best effort */ }
  }

  /** Get list of files changed since snapshot */
  static getChangedFiles(snapshot: WorkspaceSnapshot): string[] {
    try {
      if (snapshot.type === 'stash' && !snapshot.ref.startsWith('/')) {
        const output = execSync(`git diff --name-only ${snapshot.ref}`, {
          cwd: snapshot.workspacePath,
          encoding: 'utf-8',
          stdio: 'pipe',
        });
        return output.trim().split('\n').filter(Boolean);
      }
    } catch { /* ignore */ }
    return [];
  }
}
```
